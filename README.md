backend

1. cd into backend
2. first setup create virtual environment "python3 -m venv venv"
3. activate virtual environment "source venv/bin/activate"
4. install fastapi with "pip install "fastapi[standard]""
5. start server with "fastapi dev main.py"

frontend

1. cd into frontend
2. run "npm install"
3. start server with "npm run dev"
